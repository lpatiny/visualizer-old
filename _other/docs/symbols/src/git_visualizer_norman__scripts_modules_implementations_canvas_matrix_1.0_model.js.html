 10 if(typeof CI.Module.prototype._types.canvas_matrix == 'undefined')
 11 	CI.Module.prototype._types.canvas_matrix = {};

 13 CI.Module.prototype._types.canvas_matrix.Model = function(module) {
 14 	
 15 	/*
 16 	 * Sets
 17 	 * (array) this.data
 18 	 * (json) this.dataValue    -- Could be any type of data provided by any DataSource object. 
 19 	 */
 20 	CI.Module.prototype._impl.model.init(module, this);
 21 	
 22 	// Call anything else if you want. The prototyped init() function can also be used.
 23 }
 24 
 25 CI.Module.prototype._types.canvas_matrix.Model.prototype = {
 26 	
 27 	// Usually you don't really to init a model. But who knows. Please leave it.
 28 	init: function() {	
 29 		CI.Module.prototype._impl.model.afterInit(this.module);
 30 	},
 31 	
 32 	
 33 	/* 
 34 	 * This function is a handler called from any DataSource object. 
 35 	 * Its goal is to refresh the module with the new data
 36 	 * 
 37 	 * To finally refresh the view, simply call this.module.updateView();
 38 	 * or don't if for any reason it's not necessary to update the module
 39 	 */
 40 	onDataChange: function(dataName) {
 41 		/* Here you can transform the data coming from the DAO */
 42 		this.dataValue[dataName] = this.data[dataName].getData();
 43 		
 44 		/* Triggers a module update */
 45 		this.module.updateView();
 46 	},
 47 	
 48 	getValue: function() {
 49 		return this.dataValue;
 50 	}
 51 }
