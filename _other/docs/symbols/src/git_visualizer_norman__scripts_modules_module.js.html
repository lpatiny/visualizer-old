 15 /**
 16  * Creates a new base Module
 17  * @class The base class for all Modules to extend
 18  * @param {object} definition An object containing options for the grid (is merged into {@link defaults})
 19  * @param {integer} [definition.id] The id of the module
 20  * @param {string} [definition.type] A string identifying the type of this module
 21  * @param {object} [definition.size] An object definining the size of the module (containing width and height)
 22  * @param {object} [definition.position] An object definining the position of the module (containing x and y)
 23  */
 24 CI.Module = function(definition) {
 25 
 26 	this.definition = $.extend(true, {}, CI.Module.prototype.defaults, definition);
 27 	
 28 	/**
 29 	 * @function Initialises the module, constructs the DOM within the module, and initialises the MVC
 30 	 */
 31 	this.init = init;
 32 	function init() {
 33 		
 34 		//define object properties
 35 		var module = this;
 36 		var moduleType = this.definition.type;
 37 		
 38 		//Construct the DOM within the module
 39 		this.dom = $(this.buildDom());
 40 		this.domContent = this.dom.children().children('.ci-module-content').children('.ci-module-content-overflow');
 41 		this.domHeader = this.dom.children().children('.ci-module-header');
 42 		this.domWrapper = this.dom;
 43 		this.viewExpander = $('<div class="ci-module-expand">...</div>');
 44 		
 45 		//Initialises the MVC pattern for the module
 46 		this.model = new CI.Module.prototype._types[moduleType].Model(this);
 47 		this.view = new CI.Module.prototype._types[moduleType].View(this);
 48 		this.controller = new CI.Module.prototype._types[moduleType].Controller(this);
 49 		
 50 		this.view.init();
 51 		this.controller.init();
 52 		this.model.init();
 53 	}
 54 	
 55 	/**
 56 	 * Construct the basic dom for the module
 57 	 */
 58 	this.buildDom = buildDom;
 59 	function buildDom() {
 60 		
 61 		var html = [];
 62 		html.push('<div class="ci-module-wrapper ci-module-');
 63 		html.push(this.definition.type);
 64 		html.push('" data-module-id="');
 65 		html.push(this.definition.id);
 66 		html.push('"><div class="ci-module"><div class="ci-module-header"><div class="ci-module-header-title">');
 67 		html.push(this.definition.title);
 68 		html.push('</div>');
 69 		html.push('</div><div class="ci-module-content"><div class="ci-module-content-overflow">');
 70 		
 71 		html.push('</div></div>');
 72 		html.push('</div>');
 73 		return html.join('');
 74 	}
 75 	
 76 	this.init();
 77 };
 78 /**
 79  * Overrideable prototype
 80  */
 81 CI.Module.prototype = {
 82 	
 83 	/**
 84 	 * Used to define default module properties. On creation, the definition passed is merged with this object 
 85 	 * @type object
 86 	 * @hide
 87 	 */
 88 	defaults: {
 89 		id: -1,
 90 		type: 'default',
 91 		
 92 		size: {
 93 			width: 10,
 94 			height: 15
 95 		},
 96 		
 97 		position: {
 98 			top: 0,
 99 			left: 0
100 		}
101 	},
102 
103 	/**
104 	 * Contains the names of all types of module
105 	 */
106 	_types: {},
107 	
108 	
109 	/**
110 	 * Called when the data behind the module needs updating
111 	 * @param dataName The name of the piece of data being updated
112 	 * @param dataName The new value of the data being updated
113 	 */
114 	onDataChange: function(dataName, dataVal) {
115 		
116 		if(typeof this.model.onDataChange == 'function')
117 			return this.model.onDataChange(dataName, dataVal);
118 		
119 		throw "The model does not implement any dataChange function";
120 	},
121 	
122 	/**
123 	 * Called to update the view (normally after a change of data)
124 	 */
125 	updateView: function() {
126 		if(typeof this.view.update == 'function')
127 			return this.view.update();
128 			
129 		throw "The module has no update capability";
130 	},
131 	
132 	/** 
133 	 * Returns the DOM object which corresponds to the module's content
134 	 */
135 	getDomContent: function() {
136 		if(typeof this.domContent !== 'undefined')
137 			return this.domContent;
138 			
139 		throw "The module has not been loaded yet";
140 	},
141 	
142 	/** 
143 	 * Returns the DOM object which corresponds to the module's wrapper
144 	 */
145 	getDomWrapper: function() {
146 		if(typeof this.domWrapper !== 'undefined')
147 			return this.domWrapper;
148 			
149 		throw "The module has not been loaded yet";
150 	},
151 	
152 	/** 
153 	 * Returns the DOM object which corresponds to the module's view
154 	 */
155 	getDomView: function() {
156 		if(typeof this.view.getDom == 'function')
157 			return this.view.getDom();
158 			
159 		throw "The module's view doest not implement the getDom function";
160 	},
161 	
162 	/** 
163 	 * Returns the DOM object which corresponds to the module's header
164 	 */
165 	getDomHeader: function() {
166 		if(typeof this.domHeader !== 'undefined')
167 			return this.domHeader;
168 			
169 		throw "The module has not been loaded yet";
170 	},
171 	
172 	/** 
173 	 * Returns the data for the module's model
174 	 */
175 	getValue: function() {
176 		if(typeof this.model.getValue == 'function')
177 			return this.model.getValue();
178 		
179 		return;
180 	},
181 	
182 	/** 
183 	 * Returns the current position of the module
184 	 */
185 	getPosition: function() {
186 		return this.definition.position;
187 	},
188 	
189 	/** 
190 	 * Returns the current size of the module
191 	 */
192 	getSize: function() {
193 		return this.definition.size;
194 	}
195 };

197 /*
198  * Static functions
199  */
200 CI.Module.prototype._impl = {
201 	
202 	model: {
203 		
204 		/**
205 		 * Set up the model (MVC) for a module. The model contains the data concerning the module.
206 		 * @param module the module to which this model is associated
207 		 * @param model the model being associated
208 		 */
209 		init: function(module, model) {
210 			
211 			var sourceName, sourceData;
212 			
213 			model.module = module;
214 			model.data = [];
215 			model.dataValue = [];
216 			
217 			//loop through the data provided in the definition and copy it into the model as a DataSource
218 			for(var i = 0; i < module.definition.dataSource.length; i++) {
219 				sourceName = module.definition.dataSource[i].name;
220 				
221 				sourceData = null;
222 				if(typeof module.definition.dataSource[i].data !== "undefined")
223 					sourceData = module.definition.dataSource[i].data;
224 				
225 				model.data[sourceName] = new CI.DataSource(model.module, sourceName, sourceData);
226 				model.dataValue[sourceName] = null;
227 			}
228 		},
229 		
230 		/**
231 		 * Called after initialisation of the given module to bind an event to data changing, ensuring the model's data (and the module itself) is updated appropriately
232 		 */
233 		afterInit: function(module) {
234 			
235 			module.getDomWrapper().bind('sharedDataChanged', function(event, dataName, dataVal) {
236 				event.stopPropagation();
237 				event.preventDefault();
238 				
239 				module.model.data[dataName].setData(dataVal);
240 				module.onDataChange(dataName);
241 			});
242 			
243 		}
244 	},
245 	
246 	controller: {
247 		/**
248 		 * Initialise the given module's controller (mvc)
249 		 */
250 		init: function(module, controller) {
251 			
252 			controller.module = module;
253 			
254 		}
255 	}
