  2 function generateGrid(gridData, gridImage, cols, rows, cellWidth, cellHeight) {
  3 	
  4 	var gridImageData = gridImage.data;
  5 	
  6 	//var ctx = document.getElementsByTagName("canvas")[0].getContext('2d');
  7 	var color, x=0, y=0, i=0, j=0, pixelNum=0;
  8 	
  9 	while (x<cols) {
 10 		//gridData[x/cellWidth] = [];
 11 		while (y<rows) {
 12 			
 13 			
 14 			color = Math.round(gridData[x][y].value * 255);
 15 			
 16 		//	color = Math.floor( Math.random() * 255 );
 17 		//	gridData[x/cellWidth][y/cellHeight] = color;
 18 			while (i<cellWidth) {
 19 				while (j<cellHeight) {
 20 					pixelNum = 4 * (x * cellWidth + i + (y * cellHeight + j) * cols * cellWidth);
 21 					gridImageData[pixelNum+0] = color; // Red value
 22 					gridImageData[pixelNum+1] = color; // Green value
 23 					gridImageData[pixelNum+2] = color; // Blue value
 24 					gridImageData[pixelNum+3] = 255; // Alpha value
 25 					j++;
 26 				}
 27 				j=0;
 28 				i++;
 29 			}
 30 			i=0;
 31 			y++;
 32 		}
 33 		y=0;
 34 		x++;
 35 	}
 36 	
 37 	return gridImage;
 38 }
