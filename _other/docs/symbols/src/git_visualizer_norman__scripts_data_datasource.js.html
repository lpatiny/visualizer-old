  2 /**
  3  * Represents the source of any type of data (i.e. is updatable through e.g. ajax)
  4  * @Class
  5  * @namespace 
  6  * @param {Module} module the module to which this data is bound
  7  * @param {string} sourceName the name of this data object
  8  * @param {object} sourceData The actual data object (requires at least type to be set)
  9  */
 10 CI.DataSource = function(module, sourceName, sourceData) {
 11 	
 12 	this.module = module;
 13 	this.sourceName = sourceName;
 14 	this.sourceData = sourceData;
 15 	this.data = null;
 16 	
 17 	// No action required ?
 18 	if(this.sourceData == null)
 19 		return;
 20 	
 21 	if(typeof this.sourceData.type == "undefined")
 22 		return;
 23 	
 24 	
 25 	// What is the required action ?
 26 	switch(this.sourceData.type) {
 27 		
 28 		case 'url':
 29 			this.sourceUrl();
 30 		break;
 31 		
 32 		default:
 33 			
 34 		break;
 35 	}
 36 }
 37 
 38 
 39 CI.DataSource.prototype = {
 40 	
 41 	setData: function(data) {
 42 		this.data = data;
 43 	},
 44 	
 45 	getData: function() {
 46 		return this.data;
 47 	},
 48 	
 49 	getSourceData: function() {
 50 		return this.sourceData;
 51 	},
 52 	
 53 	eraseAjaxData: function() {
 54 		this.sourceData = {};
 55 	},
 56 	
 57 	setAjaxData: function(data) {
 58 		
 59 		if(typeof this.sourceData.data )
 60 		this.sourceData.data = $.extend(this.sourceData.data, data);	
 61 	},
 62 	
 63 	doGetAjaxData: function() {
 64 	
 65 		if(this.sourceData.type == "url")
 66 			return this.sourceUrl();
 67 			
 68 		throw "Impossible to get the data. DataSource does not support it"
 69 	},
 70 	
 71 	/**
 72 	 * If necessary, set up the DataSource to be updateable via Ajax (url), then run the AJAX update through jQuery
 73 	 */
 74 	sourceUrl: function() {
 75 		
 76 		if(typeof this.sourceData.url == 'undefined')
 77 			return;
 78 		
 79 		if(typeof this.sourceData.dataType == "undefined")
 80 			this.sourceData.dataType = 'html';
 81 		
 82 		if(typeof this.sourceData.method == 'undefined' || ['post', 'get'].indexOf(this.sourceData.method.toLowerCase()) == -1)
 83 			this.sourceData.method = 'get';
 84 			
 85 		if(typeof this.sourceData.data == 'undefined')
 86 			this.sourceData.data = {};
 87 			
 88 		var ajaxMethod = this.sourceData.method.toLowerCase();
 89 		
 90 		if(this.sourceData.dataType == 'json');
 91 			ajaxMethod += 'JSON';
 92 		
 93 		if(typeof jQuery[ajaxMethod] !== "function") {
 94 			throw "jQuery doesn't support ajax method : " + ajaxMethod;
 95 			return;
 96 		}	
 97 		
 98 		var source = this;
 99 		jQuery[ajaxMethod](this.sourceData.url, this.sourceData.data, function(response, xhr) {
100 			source.data = response;
101 			
102 			CI.API.setSharedVar(source.sourceName, response);
103 		});
104 	}
105 }
