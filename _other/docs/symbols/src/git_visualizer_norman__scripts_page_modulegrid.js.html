1 /**
  2  * @namespace
  3  * Represents a grid to which modules can be added, removed, moved and resized.
  4  */
  5 CI.Grid = {
  6 
  7 	/**
  8 	 * Initialise the module grid in the div with id="ci-modules-grid"
  9 	 * @param {object} definition An object containing options for the grid (is merged into {@link CI.Grid.defaults})
 10 	 * @param {integer} [definition.xWidth] The width of the grid cells
 11 	 * @param {integer} [definition.xHeight] The height of the grid cells
 12 	 */
 13 	init: function(definition) {
 14 		CI.Grid.definition = $.extend(true, CI.Grid.defaults, definition);
 15 		
 16 		CI.Grid._el = $("#ci-modules-grid");
 17 	},
 18 	
 19 	/**
 20 	 * Used to define the grid-cell sizes. On initialisation, the object passed is merged with this object 
 21 	 * @type object
 22 	 * @hide
 23 	 */
 24 	defaults: {
 25 		xWidth: 20,
 26 		xHeight: 20
 27 	},
 28 	
 29 	/**
 30 	 * Add a module to the grid 
 31 	 * @param module {Module} The module to add.
 32 	 */
 33 	addModule: function(module) {
 34 		
 35 		var modulePos = module.getPosition();
 36 		var moduleSize = module.getSize();
 37 		
 38 		module.getDomWrapper().appendTo(CI.Grid._el).css({
 39 			top: modulePos.top * CI.Grid.definition.yHeight,
 40 			left: modulePos.left * CI.Grid.definition.xWidth,
 41 			
 42 			width: moduleSize.width * CI.Grid.definition.xWidth,
 43 			height: moduleSize.height * CI.Grid.definition.yHeight
 44 		});
 45 		
 46 		
 47 		// Insert jQuery UI resizable and draggable
 48 		module.getDomWrapper().resizable({
 49 			grid: [CI.Grid.definition.xWidth, CI.Grid.definition.yHeight],
 50 			resize: function() {
 51 				CI.Grid.moduleResize(module);
 52 			},
 53 			
 54 			containment: "parent"
 55 			
 56 		}).draggable({
 57 			grid: [CI.Grid.definition.xWidth, CI.Grid.definition.yHeight],
 58 			containment: "parent"
 59 		}).trigger('resize');
 60 		
 61 		
 62 		module.getDomWrapper().on('click', '.ci-module-expand', function() {
 63 			module.getDomWrapper().height((module.getDomContent().outerHeight() + module.getDomHeader().outerHeight()));
 64 			CI.Grid.moduleResize(module);
 65 		});
 66 		
 67 		
 68 		CI.Grid.moduleResize(module);
 69 		
 70 	},
 71 	/**
 72 	 * Is called by jQuery UI when a module is resized, to resize the module and allow the module view's contents to update accordingly.
 73 	 * @param module The module to resize.
 74 	 */
 75 	moduleResize: function(module) {
 76 		
 77 		var wrapper = module.getDomWrapper();
 78 		module.getPosition().width = wrapper.width() / CI.Grid.definition.xWidth;
 79 		module.getPosition().height = wrapper.height() / CI.Grid.definition.yHeight;
 80 		
 81 		module.getDomContent().parent().css({
 82 			height: wrapper.height() - module.getDomHeader().outerHeight(true)
 83 		});
 84 		
 85 		module.view.onResize();
 86 		
 87 		CI.Grid.checkModuleSize(module);
 88 		
 89 		
 90 	},
 91 	
 92 	checkModuleSize: function(module) {
 93 		
 94 		if(module.getDomContent().height() > module.getDomContent().parent().outerHeight(true))
 95 			module.getDomContent().parent().after(module.viewExpander);
 96 		else
 97 			module.viewExpander.remove();
 98 	}
 99 }
