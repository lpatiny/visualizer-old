 10 if(typeof CI.Module.prototype._types.datamatrix_intersect == 'undefined')
 11 	CI.Module.prototype._types.datamatrix_intersect = {};
 12 
 13 CI.Module.prototype._types.datamatrix_intersect.View = function(module) {
 14 	this.module = module;
 15 }
 16 
 17 CI.Module.prototype._types.datamatrix_intersect.View.prototype = {
 18 	
 19 	init: function() {	
 20 		this.dom = $("<div />").appendTo(this.module.getDomContent());
 21 	},
 22 	
 23 	onResize: function() {
 24 		
 25 	},
 26 	
 27 	update: function() {
 28 		
 29 		html = [];
 30 		var moduleValue = this.module.getValue();
 31 		
 32 		for(var i in moduleValue) {
 33 			
 34 			html.push('<table class="ci-intersect ci-intersect-' + i + '">');
 35 			
 36 			html.push('<tr><th colspan="2">');
 37 			html.push(this.getLabel(i));
 38 			html.push('</th></tr>');
 39 			
 40 			for(var j in moduleValue[i]) {
 41 				html.push('<tr><td class="ci-label">');
 42 				html.push(this.getLabel(j));
 43 				html.push('</td><td>');
 44 				html.push(CI.toScreen(moduleValue[i][j], j, this));
 45 				html.push('</td></tr>');
 46 			}
 47 			
 48 			html.push('</table>');
 49 		}
 50 		
 51 		this.getDom().html(html.join(''));
 52 		
 53 		CI.Grid.checkModuleSize(this.module);
 54 	},
 55 	
 56 	getDom: function() {
 57 		return this.dom;
 58 	},
 59 	
 60 	getLabel: function(key) {
 61 		var val;
 62 		if(typeof (val = this.module.definition.dataModule.labels[key]) !== "undefined")
 63 			return val;
 64 			
 65 		return key;
 66 	}
 67 }
