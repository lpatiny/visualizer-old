 10 if(typeof CI.Module.prototype._types.canvas_matrix == 'undefined')
 11 	CI.Module.prototype._types.canvas_matrix = {};
 12 
 13 CI.Module.prototype._types.canvas_matrix.View = function(module) {
 14 	this.module = module;
 15 }
 16 
 17 CI.Module.prototype._types.canvas_matrix.View.prototype = {
 18 	
 19 	init: function() {	
 20 		
 21 		this.canvas = document.createElement("canvas"); // Store the pointer
 22 		this.canvasContext = this.canvas.getContext('2d'); // Store the context
 23 		this.lastCanvasWidth = 0;
 24 		this.lastCanvasHeight = 0;
 25 
 26 		this.dom = document.createElement("div");
 27 		var title = document.createElement("h3");
 28 		
 29 		this._domTitle = title;
 30 		
 31 		var table = document.createElement("table");
 32 		var tr = document.createElement("tr");
 33 		var td = document.createElement("td");
 34 		
 35 		this.dom.appendChild(title);
 36 		this.dom.appendChild(table);
 37 		
 38 		table.appendChild(tr);
 39 		tr.appendChild(td);
 40 		
 41 		td.appendChild(this.canvas);
 42 
 43 		this.module.getDomContent().html(this.dom);
 44 		
 45 		
 46 		this.worker = new Worker('./scripts/modules/implementations/canvas_matrix/1.0/worker.js');
 47 		var view = this;
 48 		this.worker.addEventListener('message', function(event) {
 49 			view.canvasContext.putImageData(event.data, 0, 0);
 50 		});
 51 		this.gridImage = this.canvasContext.createImageData(this.canvas.width, this.canvas.height);
 52 		this.updateCanvas();
 53 	},
 54 	
 55 	onResize: function() {
 56 		
 57 		var height = this.module.getDomContent().parent().height() - this.module.getDomContent().find('h3').outerHeight(true);
 58 		
 59 		var table = this.module.getDomContent().find('table').height(height);
 60 		// Set the canvas to full width and height
 61 		var moduleWidth = table.width();
 62 		var moduleHeight = table.height();
 63 		
 64 		var size = Math.min(moduleWidth, moduleHeight);
 65 		
 66 		if(this.rowNumber == undefined || this.colNumber == undefined)
 67 			return;
 68 		
 69 		this.cellHeight = Math.floor(size / this.rowNumber);
 70 		this.cellWidth = Math.floor(size / this.colNumber); 
 71 		
 72 		var newWidth = this.cellWidth * this.colNumber;
 73 		var newHeight = this.cellHeight * this.rowNumber;
 74 		
 75 		
 76 		if(newWidth != this.lastCanvasWidth || newHeight != this.lastCanvasHeight) {
 77 			
 78 			this.lastCanvasWidth = this.canvas.width;
 79 			this.lastCanvasHeight = this.canvas.height;
 80 			
 81 			if(newWidth == 0 || newHeight == 0)
 82 				return;
 83 			
 84 			this.canvas.width = newWidth;
 85 			this.canvas.height = newHeight;
 86 			
 87 			this.gridImage = this.canvasContext.createImageData(this.canvas.width, this.canvas.height); // Store the image
 88 			this.updateCanvas();
 89 		}
 90 	},
 91 	
 92 	update: function() {
 93 		
 94 		var moduleValue = this.module.getValue();
 95 		for(var i in moduleValue) {
 96 			
 97 			this.colNumber = moduleValue[i].nbCols;
 98 			this.rowNumber = moduleValue[i].nbRows;
 99 			
100 			this._domTitle.innerHTML = moduleValue[i].data.title;
101 			
102 			break;
103 		}
104 		
105 		
106 		
107 		this.onResize()
108 	},
109 	
110 	updateCanvas: function() {
111 		
112 		var view = this,
113 		    moduleValue = this.module.getValue();
114 		
115 		for(var i in moduleValue) {
116 			
117 			if(moduleValue[i] == null || this.gridImage == undefined)
118 				continue;
119 			
120 			this.worker.postMessage({
121 				gridData: moduleValue[i].dataMatrix,
122 				gridImageData: this.gridImage,
123 				cols: this.colNumber,
124 				rows: this.rowNumber,
125 				cellWidth: this.cellWidth,
126 				cellHeight: this.cellHeight
127 			});
128 		
129 			break;
130 		}
131 		
132 	},
133 	
134 	getDom: function() {
135 		return this.dom;
136 	},
137 	
138 	// No additional type transform needed
139 	typeToScreen: {}
140 }
